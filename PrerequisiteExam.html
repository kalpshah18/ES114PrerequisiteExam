def fibonacciList(n: int) -> list:
    ls = [0, 1]
    if n == 1:
        return [0]
    for i in range(n - 2):
        ls.append(ls[-1] + ls[-2])
    return ls

Question 9: Merging Dictionaries

We cannot use '|' Operator Because '|' Operator overwrites Value for Same Key with Value of Second Dictionary. Thus, we have to iterate over each item of both dictionaries.

In[57]:
def mergeDictionaries(dict1: dict, dict2: dict) -> dict:
    result = {}
    for i in dict1.items():
        result[i[0]] = result.get(i[0], 0) + i[1]
    for i in dict2.items():
        result[i[0]] = result.get(i[0], 0) + i[1]
    return result

# Example
dict1 = {'a': 10, 'b': 20}
dict2 = {'b': 5, 'c': 15}
print("Dictionary 1:", dict1)
print("Dictionary 2:", dict2)
print("Merged Dictionary", mergeDictionaries(dict1, dict2))

Question 11: Remove Duplicates

As of Python 3.7, Dictionaries have become ordered. We will use this Property to Answer the Question

In[9]:
def removeDuplicates(l: list) -> list:
    d = {}
    for element in l:
        d[element] = d.get(element, 0) + 1
    return list(d.keys())

def mean(l:list) -> float:
    s = sum(l)
    n = len(l)
    return s / n

def median(l: list):
    l.sort()
    n = len(l)
    if n % 2 == 1:
        return l[n // 2] # Indices start from 0
    else:
        n1 = l[n // 2 - 1] # Indices start from 0 so we had to subtract 1
        n2 = l[n // 2]
        return (n1 + n2) / 2

Question 15: Anagram Finder

We will Sort the Letters Alphabetically then for all Words, check the Sorted Version of those Words and if they match then they are Anagrams.

In[37]:
def anagramFinder(word: str, ls: [str]) -> [str]:
    output = []
    sortedWord = sorted(word)
    for w in ls:
        if sorted(w) == sortedWord:
            output.append(w)
    return output

from matplotlib_venn import venn2

def setOperations(s1, s2):
    intersection = s1.intersection(s2)
    union = s1.union(s2)
    symmetric_difference = s1.symmetric_difference(s2)
    return intersection, union, symmetric_difference

def visualizeResults(ls1, ls2):
    s1 = set(ls1)
    s2 = set(ls2)
    
    results = list(setOperations(s1, s2))
    print("Intersection:", results[0])
    print("Union:", results[1])
    print("Symmetric Difference:", results[2])
    
    venn2([s1, s2], set_labels=('List 1', 'List 2'))
    plt.title('Set Operations Visualization')
    plt.show()
    plt.close()

def fileOperations(ls: [int]) -> int:
    with open('data.txt', 'wt') as fs:
        for num in ls:
            fs.write(str(num) + '\n')
    fs.close()
    with open('data.txt', 'rt') as fsRead:
        s = 0
        while True:
            try:
                s += int(fsRead.readline())
            except:
                break
        return s

def statistics(ls: [int]):
    s = sum(ls)
    n = len(ls)
    mean = s / n

    median = 0
    ls.sort()
    if n % 2 == 1:
        median = ls[n // 2] # Indices start from 0
    else:
        n1 = l[n // 2 - 1] # Indices start from 0 so we had to subtract 1
        n2 = l[n // 2]
        median = (n1 + n2) / 2

    frequency = {}
    for num in ls:
        frequency[num] = frequency.get(num, 0) + 1
    maxVal = 0
    mode = 0
    for key, value in frequency.items():
        if value > maxVal:
            maxVal = value
            mode = key

    stdDevSum = 0
    for num in ls:
        stdDevSum += abs(num - mean)
    stdDeviation = stdDevSum / n
    return mean, median, mode, stdDeviation

    s = sum(ls)
